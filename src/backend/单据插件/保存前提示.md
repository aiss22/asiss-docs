# 单据插件：保存前提示

`BeforeSave` 使用保存前事件

 ```1
 //保存前提示（是否保存）
 private bool doSave = false;
 public override void BeforeSave(BeforeSaveEventArgs e)
 {
     base.BeforeSave(e);
     //否定保存等待确认
     if(doSave)
     {
         doSave = false;
         return;
     }
     e.Cancel = true;//禁止
     this.View.ShowMessage("请确认是否保存！",MessageBoxOptions.OKCancel,result => 
     {
         //点击OK后
         if(result == MessageBoxResult.OK) 
         {
             //继续保存操作
             doSave=true;
             this.View.InvokeFormOperation("Save");//调用表单动作'保存'
         }
     });

 }
 ```
