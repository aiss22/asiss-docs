# 常用的消息提示框

`this.View.ShowMessage("你好");`//常用消息弹窗

`this.View.ShowErrMessage("严重错误");`//错误信息弹窗

//询问式消息框

```示例
this.View.ShowMessage("错误提示，是否继续？"
  ,MessageBoxOptions.YesNoCancel,
  result =>
  {
      if(result == MessageBoxResult.Yes)
      {
          //选择“是”给备注赋值“是”
          this.View.Model.SetValue("FNote","是");
      }
      else if(result == MessageBoxResult.No)
      {
          //选择否，给备注赋值否
          this.View.Model.SetValue("FNote","否");
      }
      else if(result == MessageBoxResult.Cancel)
      {
          //选择取消，给备注赋值取消
          this.View.Model.SetValue("FNote","取消");
      }
  });
  //刷新备注
  this.View.UpdateView("FNote");
```

//警告式消息框

```示例
this.View.ShowWarnningMessage("不能对其操作，请确认",
    action: (result) =>
    {
        //关闭页面
        this.View.Close();
    });
```
