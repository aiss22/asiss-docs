# 构造函数

```示例
public class Car
{
    public string model;
    public string color;

    //创建带参构造函数
    public Car(string modelName,string modelColor) 
    {
        model= modelName;
        color= modelColor;
    }
}
public class Program
{
    static void Main(string[] args)
    {
        Car car = new Car("小米","红色");
        Console.WriteLine(car.model + car.color);
    }
}
```
